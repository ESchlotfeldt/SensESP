109void Networking::setup_saved_ssid() {
110 this->emit(WiFiState::kWifiDisconnected);
111 setup_wifi_callbacks();
112
113 String hostname = SensESPBaseApp::get_hostname();
114 WiFi.setHostname(hostname.c_str());
115
116 if (ap_mode_ == false) {
117 // set up WiFi in regular STA (client) mode
118 auto reconnect_cb = [this]() {
119 if (WiFi.status() != WL_CONNECTED) {
120 debugI("Connecting to wifi SSID %s.", ap_ssid.c_str());
121 WiFi.begin(ap_ssid.c_str(), ap_password.c_str());
122 }
123 };
124
125 // Perform an initial connection without a delay.
126 reconnect_cb();
127
128 // Launch a separate onRepeat reaction to (re-)establish WiFi connection.
129 // Connecting is attempted only every 20 s to allow the previous connection
130 // attempt to complete even if the network is slow.
131 ReactESP::app->onRepeat(20000, reconnect_cb);
132 } else {
133 // Set up WiFi in AP mode. In this case, we don't need a reconnect loop.
134 debugI("Setting up a WiFi access point...");
135 WiFi.softAP(ap_ssid.c_str(), ap_password.c_str());
136 }
137}

177void Networking::setup_wifi_manager() {
178 wifi_manager = new AsyncWiFiManager(server, dns);
179
180 String hostname = SensESPBaseApp::get_hostname();
181
182 setup_wifi_callbacks();
183
184 // set config save notify callback
185 wifi_manager->setBreakAfterConfig(true);
186
187 wifi_manager->setConfigPortalTimeout(WIFI_CONFIG_PORTAL_TIMEOUT);
188
189#ifdef SERIAL_DEBUG_DISABLED
190 wifi_manager->setDebugOutput(false);
191#endif
192 AsyncWiFiManagerParameter custom_hostname("hostname", "Device hostname",
193 hostname.c_str(), 64);
194 wifi_manager->addParameter(&custom_hostname);
195
196 AsyncWiFiManagerParameter custom_ap_ssid(
197 "ap_ssid", "Custom Access Point SSID", ap_ssid.c_str(), 33);
198 wifi_manager->addParameter(&custom_ap_ssid);
199
200 AsyncWiFiManagerParameter custom_ap_password(
201 "ap_password", "Custom Access Point Password", ap_password.c_str(), 64);
202 wifi_manager->addParameter(&custom_ap_password);
203
204 wifi_manager->setTryConnectDuringConfigPortal(false);
205
206 // Create a unique SSID for configuring each SensESP Device
207 String config_ssid;
208 if (wifi_manager_ap_ssid_ != "") {
209 config_ssid = wifi_manager_ap_ssid_;
210 } else {
211 config_ssid = "Configure " + hostname;
212 }
213 const char* pconfig_ssid = config_ssid.c_str();
214
215 this->emit(WiFiState::kWifiManagerActivated);
216
217 WiFi.setHostname(SensESPBaseApp::get_hostname().c_str());
218
219 wifi_manager->startConfigPortal(pconfig_ssid, wifi_manager_password_);
220
221 String configured_custom_ap_ssid = custom_ap_ssid.getValue();
222 String configured_custom_ap_password = custom_ap_password.getValue();
223
224 bool connected = false;
225
226 if (configured_custom_ap_ssid != "" && configured_custom_ap_password != "") {
227 // AP mode is desired
228 ap_ssid = configured_custom_ap_ssid;
229 ap_password = configured_custom_ap_password;
230 ap_mode_ = true;
231
232 // always assume we can launch a soft AP
233 connected = true;
234 } else {
235 // WiFiManager attempts to connect to the new SSID, but that doesn't seem to
236 // work reliably. Instead, we'll just attempt to connect manually.
237
238 this->ap_ssid = wifi_manager->getConfiguredSTASSID();
239 this->ap_password = wifi_manager->getConfiguredSTAPassword();
240
241 // attempt to connect with the new SSID and password
242 if (this->ap_ssid != "" && this->ap_password != "") {
243 debugD("Attempting to connect to acquired SSID %s and password",
244 this->ap_ssid.c_str());
245 WiFi.begin(this->ap_ssid.c_str(), this->ap_password.c_str());
246 for (int i = 0; i < 20; i++) {
247 if (WiFi.status() == WL_CONNECTED) {
248 connected = true;
249 break;
250 }
251 delay(1000);
252 }
253 }
254 }
255
256 // Only save the new configuration if we were able to connect to the new SSID.
257
258 if (connected) {
259 String new_hostname = custom_hostname.getValue();
260 debugI("Got new custom hostname: %s", new_hostname.c_str());
261 SensESPBaseApp::get()->get_hostname_observable()->set(new_hostname);
262 debugI("Got new SSID and password: %s", ap_ssid.c_str());
263 save_configuration();
264 }
265 debugW("Restarting...");
266 ESP.restart();
267}
